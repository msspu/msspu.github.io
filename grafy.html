Vývoj zaměstnanecké struktury 1995 až 2018

načtení agregovaných dat, obsahují podíly celkové pracovní síly v ČR za typy ekonomické aktivity:

- Zaměstnanci na plný úvazek v kombinaci se smlouvou na dobu neurčitou (tz. standardní úvazky)

- Podnikatelé se zaměstnanci

- OSVČ

- Kombinace nestandardních úvazků

  • částečný úvazek na dobu neurčitou

  • částečný úvazek na dobu určitou

  • plný úvazek na dobu určitou

library(readxl)
library(tidyr)
library(dplyr)
dataPost <- read_excel("h:/moje/DATA/Post95to18.xlsx")
dataPost$year <- as.Date(paste(dataPost$REFTROK, 1, 1, sep = "-"))

Transformace dat

df <- dataPost %>%
  select(year, CastNeurcita, PlnyUrcita, CastUrcita, Nezam, OSVC ) %>%
  gather(key = "typ", value = "value", -year)

dataPost$Self <- (dataPost$Podnikatel + dataPost$OSVC)
dataPost$Nons <- (dataPost$CastNeurcita + dataPost$PlnyUrcita + dataPost$CastUrcita )

df2 <- dataPost %>%
  select(year, Self, Nons, Zam) %>%
  gather(key = "typ", value = "value", -year)


df$typ <- recode(df$typ, CastNeurcita = "Částečný úvazek na dobu neurčitou", PlnyUrcita = "Plný úvazek na dobu určitou", CastUrcita = "Částečný úvazek na dobu určitou", Nezam = "Nezaměstnanost", OSVC = "OSVČ"  )

df2$typ <- recode(df2$typ, Self = "Podnikatelé/OSVČ", Nons = "Nestandardní úvazky", Zam = "Zaměstnanci na plný úvazek a dobu neurčitou")
library(ggplot2)
ggplot(df2, aes(x = year, y = value)) + 
  geom_line(aes(color = typ), size = 1) +
    theme_minimal()+
 labs(
    x = "",
    y = "% z pracovní síly",
    title = "Ekonomická aktivita 1995 - 2018",
    subtitle = ""
  )

ggplot(df, aes(x = year, y = value)) +

 labs(
